used the same structure for all question types 
only differce is when it came to true false question we won't need to use same parts as answers , because we already now 
the answer is true or false , so we igonre variable of answer_1 2 3,
for question type :  1 for True/False , 2 for Single Answer , 3 for Multi Answer 
struct Question {
    // Variable that defines type of question :: 1 for True/False , 2 for Single Answer , 3 for Multi Answer
    int type ;
    // setting max count of question text string
    char text[512];
    // array of answer (string) , for True False we won't use this array
    char answer_1[256];
    char answer_2[256];
    char answer_3[256];
    
    // Points for each answer , for True False we use only emlemnts of array 
    int points[3];
};
